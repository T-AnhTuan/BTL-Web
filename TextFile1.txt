[HttpGet] // Hoặc [HttpPost] tùy theo giao diện nút Duyệt của bạn
public async Task<IActionResult> DuyetPhieuNhap(int id)
{
    // 1. Tìm phiếu và kèm theo TẤT CẢ chi tiết vật tư bên trong (Rất quan trọng, thiếu dòng Include là tồn kho = 0)
    var phieuNhap = await _context.PhieuNhaps
        .Include(p => p.ChiTietPhieuNhaps)
        .FirstOrDefaultAsync(p => p.Id == id);

    if (phieuNhap == null)
    {
        TempData["ErrorMsg"] = "Không tìm thấy phiếu nhập.";
        return RedirectToAction("Index");
    }

    // Chặn nếu phiếu đã duyệt rồi thì không cộng kho thêm lần nữa
    if (phieuNhap.TrangThai == TrangThaiPhieuNhap.DaDuyet)
    {
        TempData["ErrorMsg"] = "Phiếu này đã được duyệt trước đó rồi!";
        return RedirectToAction("XemChiTiet", new { id = phieuNhap.Id });
    }

    // 2. Dùng Transaction để đảm bảo: Đã cộng là phải lưu thành công cả cụm
    using var transaction = await _context.Database.BeginTransactionAsync();
    try
    {
        // Đổi trạng thái phiếu
        phieuNhap.TrangThai = TrangThaiPhieuNhap.DaDuyet;

        // 3. VÒNG LẶP CỘNG TRỰC TIẾP TỒN KHO VÀO BẢNG VẬT TƯ
        foreach (var chiTiet in phieuNhap.ChiTietPhieuNhaps)
        {
            var vatTu = await _context.VatTus.FirstOrDefaultAsync(v => v.Id == chiTiet.VatTuId);
            
            if (vatTu != null)
            {
                // Tính toán giá vốn (Bình quân gia quyền)
                decimal tongGiaTriCu = vatTu.TonKhoHienTai * vatTu.GiaVonBinhQuan;
                decimal tongGiaTriMoi = chiTiet.SoLuong * chiTiet.DonGiaNhap;

                // CỘNG DỒN SỐ LƯỢNG THỰC TẾ
                long tonKhoMoi = vatTu.TonKhoHienTai + chiTiet.SoLuong;

                if (tonKhoMoi > 0)
                {
                    vatTu.GiaVonBinhQuan = (tongGiaTriCu + tongGiaTriMoi) / tonKhoMoi;
                }

                // Lưu lại vào DB
                vatTu.TonKhoHienTai = tonKhoMoi;
                _context.VatTus.Update(vatTu);
            }
        }

        // 4. Lưu toàn bộ thay đổi và xác nhận (Commit)
        await _context.SaveChangesAsync();
        await transaction.CommitAsync();

        TempData["SuccessMsg"] = "Đã duyệt phiếu và CỘNG TỒN KHO thành công 100%!";
        return RedirectToAction("XemChiTiet", new { id = phieuNhap.Id });
    }
    catch (Exception ex)
    {
        await transaction.RollbackAsync();
        TempData["ErrorMsg"] = "Lỗi nghiêm trọng khi cộng tồn kho: " + ex.Message;
        return RedirectToAction("XemChiTiet", new { id = phieuNhap.Id });
    }
}
